🧠 5. PROGRESSION SYSTEM

📁 modes/progression/config/progression.yaml

This is your backbone.

mode:
  start_events: ball_started

variable_player:
  villain_completed:
    progression_score:
      int: 1
      action: add

event_player:

  progression_score{value>=3}:
    - start_mini_wizard

  progression_score{value>=6}:
    - enable_multiball

  progression_score{value>=9}:
    - start_kingpin
